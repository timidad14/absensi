🗄️ Panduan Setup Supabase
Klik untuk lihat SQL — jalankan sekali di SQL Editor Supabase
▶ 
-- 1. Tabel data guru
CREATE TABLE IF NOT EXISTS guru (
  id          BIGSERIAL PRIMARY KEY,
  id_pps      TEXT UNIQUE NOT NULL,
  nama        TEXT NOT NULL,
  dom         TEXT,
  tingkatan   TEXT,
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- 2. Tabel absensi harian
CREATE TABLE IF NOT EXISTS absensi (
  id          BIGSERIAL PRIMARY KEY,
  id_pps      TEXT NOT NULL REFERENCES guru(id_pps) ON DELETE CASCADE,
  tanggal     DATE NOT NULL,
  status      TEXT NOT NULL CHECK (status IN ('H','A')),
  keterangan  TEXT,
  created_at  TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(id_pps, tanggal)
);

-- 3. Index untuk performa
CREATE INDEX IF NOT EXISTS idx_absensi_tanggal ON absensi(tanggal);
CREATE INDEX IF NOT EXISTS idx_absensi_idpps   ON absensi(id_pps);

-- 4. Aktifkan RLS — WAJIB agar bisa diakses dari browser!
ALTER TABLE guru ENABLE ROW LEVEL SECURITY;
ALTER TABLE absensi ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public read guru"    ON guru    FOR SELECT USING (true);
CREATE POLICY "public write guru"   ON guru    FOR ALL    USING (true);
CREATE POLICY "public read absensi" ON absensi FOR SELECT USING (true);
CREATE POLICY "public write absensi" ON absensi FOR ALL   USING (true);
